from js import document def get(id): return float(document.getElementById(id).value) def tambah(): document.getElementById("hasil").innerText = "Hasil: " + str(get("a") + get("b")) def kurang(): document.getElementById("hasil").innerText = "Hasil: " + str(get("a") - get("b")) def kali(): document.getElementById("hasil").innerText = "Hasil: " + str(get("a") * get("b")) def bagi(): if get("b") == 0: document.getElementById("hasil").innerText = "Tidak bisa dibagi 0" else: document.getElementById("hasil").innerText = "Hasil: " + str(get("a") / get("b"))
